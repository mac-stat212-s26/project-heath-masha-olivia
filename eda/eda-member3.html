Summary Statistics

Code 
stats_tbl <- states %>%
  select(-location) %>%
  pivot_longer(everything(), names_to = "coverage_type", values_to = "value") %>%
  group_by(coverage_type) %>%
  summarise(Mean   = mean(value),
    Median = median(value),
    SD     = sd(value),
    Min    = min(value),
    Max    = max(value),
    .groups = "drop") %>% 
  mutate(coverage_type = str_to_title(str_replace_all(coverage_type, "_", " ")),
    across(where(is.numeric), ~ percent(.x, accuracy = 0.1))) %>%
  arrange(desc(Mean)) %>%
  rename("Coverage Type" = coverage_type)

knitr::kable(stats_tbl, align = "lrrrrr",
             caption = "Summary statistics by coverage type (states & territories)")

make_ranked_plot <- function(col, n = 10, title, fill_col) {
  states |>
    select(location, value = {{ col }}) |>
    slice_max(value, n = n) |>
    mutate(location = fct_reorder(location, value)) |>
    ggplot(aes(x = value, y = location)) +
    geom_col(fill = fill_col, alpha = 0.85) +
    geom_text(aes(label = percent(value, accuracy = 0.1)),
              hjust = -0.1, size = 3.2) +
    scale_x_continuous(labels = percent_format(accuracy = 1),
                       expand = expansion(mult = c(0, 0.15))) +
    labs(title = title, x = NULL, y = NULL)
}

p3a <- make_ranked_plot(uninsured, 10, "Top 10: Highest Uninsured Rate",   "#E74C3C")
p3b <- make_ranked_plot(employer,  10, "Top 10: Highest Employer Coverage", "#2980B9")

p3a | p3b

Correlation Between Coverage Types

Code 
cor_mat <- states %>%
  select(-location) %>%
  cor() %>%
  as.data.frame() %>%
  rownames_to_column("var1") %>%
  pivot_longer(-var1, names_to = "var2", values_to = "corr") %>%
  mutate(
    var1 = str_to_title(str_replace_all(var1, "_", " ")),
    var2 = str_to_title(str_replace_all(var2, "_", " "))
  )

ggplot(cor_mat, aes(x = var1, y = var2, fill = corr)) +
  geom_tile(color = "white", linewidth = 0.5) +
  geom_text(aes(label = round(corr, 2)), size = 3.5) +
  scale_fill_gradient2(low = "#E74C3C", mid = "white", high = "#2980B9",
                       midpoint = 0, limits = c(-1, 1),
                       name = "Pearson r") +
  labs(
    title    = "Correlation Matrix: Coverage Types",
    subtitle = "Computed across all states and territories",
    x = NULL, y = NULL
  ) +
  theme(axis.text.x = element_text(angle = 30, hjust = 1))

Notable correlations:

  • Employer ↔︎ Medicaid: strong negative (-0.75) — states with higher employer coverage tend to have lower Medicaid enrollment.
  • Medicaid ↔︎ Uninsured: negative (-0.21) — expanded Medicaid is associated with fewer uninsured residents.
